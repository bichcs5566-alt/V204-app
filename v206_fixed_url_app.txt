let AUTO_TIMER = null;
let RAW_NAV = [];
let RAW_POS = [];
let RAW_SUM = [];

function parseCSV(text){
  const rows=[]; let row=[], cur="", inQuotes=false;
  for(let i=0;i<text.length;i++){
    const ch=text[i], next=text[i+1];
    if(ch === '"'){
      if(inQuotes && next === '"'){ cur += '"'; i++; } else inQuotes = !inQuotes;
    } else if(ch === ',' && !inQuotes){
      row.push(cur); cur="";
    } else if((ch === '\n' || ch === '\r') && !inQuotes){
      if(ch === '\r' && next === '\n') i++;
      row.push(cur);
      if(row.length > 1 || row[0] !== "") rows.push(row);
      row=[]; cur="";
    } else cur += ch;
  }
  if(cur.length || row.length){ row.push(cur); rows.push(row); }
  if(!rows.length) return [];
  const headers = rows[0].map(h=>h.trim());
  return rows.slice(1).map(r => {
    const o={}; headers.forEach((h,i)=>o[h]=(r[i]??"").trim()); return o;
  });
}
function toNum(x){ const n = Number(String(x ?? "").replace(/,/g,"")); return Number.isFinite(n)?n:NaN; }
function fmtNum(x,d=2){ const n=toNum(x); return Number.isFinite(n)?n.toLocaleString(undefined,{minimumFractionDigits:d,maximumFractionDigits:d}):"-"; }
function fmtPct(x,d=2){ const n=toNum(x); return Number.isFinite(n)?(n*100).toFixed(d)+"%":"-"; }
function nowStamp(){ return new Date().toLocaleString(); }
function safeDate(x){ return x ? String(x).slice(0,10) : "-"; }

function renderTable(id, rows){
  const el=document.getElementById(id);
  if(!rows?.length){ el.innerHTML="<tr><td>目前沒有資料</td></tr>"; return; }
  const cols=Object.keys(rows[0]);
  let html="<thead><tr>"+cols.map(c=>`<th>${c}</th>`).join("")+"</tr></thead><tbody>";
  for(const r of rows){ html += "<tr>"+cols.map(c=>`<td>${r[c] ?? ""}</td>`).join("")+"</tr>"; }
  html += "</tbody>";
  el.innerHTML=html;
}
function drawLineChart(canvasId, labels, values){
  const canvas=document.getElementById(canvasId), ctx=canvas.getContext("2d"), W=canvas.width, H=canvas.height;
  ctx.clearRect(0,0,W,H); ctx.fillStyle="#0c121b"; ctx.fillRect(0,0,W,H);
  if(!values || values.length < 2){ ctx.fillStyle="#95a3b3"; ctx.font="14px sans-serif"; ctx.fillText("目前沒有足夠資料可畫圖",20,30); return; }
  const pad={l:42,r:12,t:16,b:26}, minV=Math.min(...values), maxV=Math.max(...values), range=(maxV-minV)||1;
  ctx.strokeStyle="#273242"; ctx.lineWidth=1;
  for(let i=0;i<4;i++){ const y=pad.t+(H-pad.t-pad.b)*(i/3); ctx.beginPath(); ctx.moveTo(pad.l,y); ctx.lineTo(W-pad.r,y); ctx.stroke(); }
  ctx.strokeStyle="#7cb7ff"; ctx.lineWidth=2.4; ctx.beginPath();
  values.forEach((v,i)=>{ const x=pad.l+(W-pad.l-pad.r)*(i/(values.length-1)); const y=H-pad.b-((v-minV)/range)*(H-pad.t-pad.b); if(i===0) ctx.moveTo(x,y); else ctx.lineTo(x,y); });
  ctx.stroke();
}
function setErr(msg){ document.getElementById("errorBox").textContent = msg || ""; }
function setChangeHint(msg){ document.getElementById("changeHint").textContent = msg || ""; }
function setBaseDisplay(base){ document.getElementById("baseUrlText").textContent = base || "未設定"; document.getElementById("baseUrlInput").value = base || ""; }
function priceTierLabel(v){
  return {all:"全部",lt10:"10 以下","10_30":"10–30","30_50":"30–50","50_100":"50–100","100p":"100 以上",unknown:"未知"}[v] || "全部";
}
function refreshLabel(v){ return v === "0" ? "關閉" : `每 ${v} 分鐘`; }

function inferPrice(row){
  const keys = ["price","entry_price","close","close_used","last_price","px"];
  for(const k of keys){
    const n = toNum(row[k]);
    if(Number.isFinite(n)) return n;
  }
  return NaN;
}
function matchTier(price, tier){
  if(tier === "all") return true;
  if(!Number.isFinite(price)) return tier === "unknown";
  if(tier === "lt10") return price < 10;
  if(tier === "10_30") return price >= 10 && price < 30;
  if(tier === "30_50") return price >= 30 && price < 50;
  if(tier === "50_100") return price >= 50 && price < 100;
  if(tier === "100p") return price >= 100;
  return true;
}

function buildDashboard(navRows, posRows, sumRows){
  const summary = sumRows[0] || {};
  const latestNav = navRows[navRows.length-1] || {};

  document.getElementById("m_nav").textContent = fmtNum(latestNav.nav ?? summary.final_nav,0);
  document.getElementById("m_total").textContent = fmtPct(summary.total_return,2);
  document.getElementById("m_sharpe").textContent = fmtNum(summary.sharpe,2);
  document.getElementById("m_mdd").textContent = fmtPct(summary.mdd,2);
  document.getElementById("m_holdings").textContent = fmtNum(summary.avg_holdings,2);
  document.getElementById("m_exposure").textContent = fmtNum(summary.avg_exposure,2);
  document.getElementById("lastLoadedAt").textContent = localStorage.getItem("v206_loaded_at") || "-";

  drawLineChart("navChart", navRows.map(r=>safeDate(r.date)), navRows.map(r=>toNum(r.nav)).filter(Number.isFinite));

  const maxSignal = posRows?.length ? [...posRows].sort((a,b)=>String(a.signal_date||"").localeCompare(String(b.signal_date||""))).slice(-1)[0].signal_date : null;
  const signalRows = maxSignal ? posRows.filter(r=>r.signal_date===maxSignal) : [];
  renderTable("signalTable", signalRows.slice(0,50));

  const maxTrade = posRows?.length ? [...posRows].sort((a,b)=>String(a.trade_date||"").localeCompare(String(b.trade_date||""))).slice(-1)[0].trade_date : null;
  const tradeRows = maxTrade ? posRows.filter(r=>r.trade_date===maxTrade) : [];
  renderTable("tradeTable", tradeRows.slice(0,50));

  renderTable("summaryTable", [summary]);
}

function applyFilters(){
  const tier = localStorage.getItem("v206_price_tier") || "all";
  document.getElementById("priceTierText").textContent = priceTierLabel(tier);
  const filteredPos = RAW_POS.filter(r => matchTier(inferPrice(r), tier));
  buildDashboard(RAW_NAV, filteredPos, RAW_SUM);
}

async function fetchText(url){
  const r = await fetch(url, {cache:"no-store"});
  if(!r.ok) throw new Error(`${url} 讀取失敗：HTTP ${r.status}`);
  return await r.text();
}
function csvSignature(sumText){ return btoa(unescape(encodeURIComponent(sumText))).slice(0,80); }

async function refreshFromUrl(auto=false){
  setErr("");
  const base = (localStorage.getItem("v206_base_url") || "").trim().replace(/\/+$/,'');
  setBaseDisplay(base);
  if(!base){ setErr("請先設定 Base URL。"); return; }

  try{
    const navText = await fetchText(base + "/v202_nav.csv");
    const posText = await fetchText(base + "/v202_positions.csv");
    const sumText = await fetchText(base + "/v202_summary.csv");

    const oldSig = localStorage.getItem("v206_summary_sig") || "";
    const newSig = csvSignature(sumText);

    RAW_NAV = parseCSV(navText);
    RAW_POS = parseCSV(posText);
    RAW_SUM = parseCSV(sumText);

    localStorage.setItem("v206_nav_text", navText);
    localStorage.setItem("v206_pos_text", posText);
    localStorage.setItem("v206_sum_text", sumText);
    localStorage.setItem("v206_summary_sig", newSig);
    localStorage.setItem("v206_loaded_at", nowStamp());

    applyFilters();

    if(oldSig && oldSig !== newSig){
      setChangeHint("偵測到新資料，已自動更新。");
    }else{
      setChangeHint(auto ? "自動檢查完成，暫無新變化。" : "資料已同步，暫無新變化。");
    }
  }catch(err){
    setErr("自動抓取失敗：\n" + err.message);

    const navText = localStorage.getItem("v206_nav_text");
    const posText = localStorage.getItem("v206_pos_text");
    const sumText = localStorage.getItem("v206_sum_text");
    if(navText && posText && sumText){
      RAW_NAV = parseCSV(navText);
      RAW_POS = parseCSV(posText);
      RAW_SUM = parseCSV(sumText);
      applyFilters();
    }
  }
}

function saveBaseUrl(){
  const v = document.getElementById("baseUrlInput").value.trim();
  localStorage.setItem("v206_base_url", v);
  setBaseDisplay(v);
  setChangeHint("網址已儲存。");
}
function setupAutoRefresh(){
  if(AUTO_TIMER) clearInterval(AUTO_TIMER);
  const v = localStorage.getItem("v206_auto_refresh") || "5";
  document.getElementById("autoRefreshSelect").value = v;
  document.getElementById("autoRefreshText").textContent = refreshLabel(v);
  if(v !== "0"){
    AUTO_TIMER = setInterval(() => refreshFromUrl(true), Number(v) * 60 * 1000);
  }
}

document.getElementById("saveBaseBtn").addEventListener("click", saveBaseUrl);
document.getElementById("refreshBtn").addEventListener("click", () => refreshFromUrl(false));
document.getElementById("autoRefreshSelect").addEventListener("change", (e)=>{
  localStorage.setItem("v206_auto_refresh", e.target.value);
  setupAutoRefresh();
});
document.getElementById("priceTierSelect").addEventListener("change", (e)=>{
  localStorage.setItem("v206_price_tier", e.target.value);
  applyFilters();
});

setBaseDisplay(localStorage.getItem("v206_base_url") || "");
document.getElementById("priceTierSelect").value = localStorage.getItem("v206_price_tier") || "all";
setupAutoRefresh();

const navText = localStorage.getItem("v206_nav_text");
const posText = localStorage.getItem("v206_pos_text");

}
