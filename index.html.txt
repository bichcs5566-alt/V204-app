<script>

const tierMap = {
    all: "tier_all_positions.csv",
    lt10: "tier_lt10_positions.csv",
    "10_30": "tier_10_30_positions.csv",
    "30_50": "tier_30_50_positions.csv",
    "50_100": "tier_50_100_positions.csv",
    "100p": "tier_100p_positions.csv"
};

async function loadSummary(){
    try{
        const res = await fetch("tier_overview_summary.csv");
        const text = await res.text();
        document.getElementById("summary").innerText = text;
    }catch(e){
        document.getElementById("summary").innerText = "summary讀取失敗";
    }
}

async function loadTable(tier){
    const file = tierMap[tier];
    try{
        const res = await fetch(file);
        const text = await res.text();

        const rows = text.split("\n").slice(1);
        const tbody = document.querySelector("#table tbody");
        tbody.innerHTML = "";

        rows.forEach(r=>{
            if(!r) return;
            const cols = r.split(",");

            const tr = document.createElement("tr");

            const pnl = parseFloat(cols[3] || 0);

            tr.innerHTML = `
            <td>${cols[0]}</td>
            <td>${cols[1]}</td>
            <td>${cols[2]}</td>
            <td class="${pnl>=0?'good':'bad'}">${pnl}</td>
            <td>${cols[4]}</td>
            `;

            tbody.appendChild(tr);
        });

    }catch(e){
        console.log(e);
    }
}

document.getElementById("tierSelect").addEventListener("change", (e)=>{
    loadTable(e.target.value);
});

loadSummary();
loadTable("all");

</script>
